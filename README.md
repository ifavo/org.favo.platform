org.favo.platform
=================

A little commonjs helper for Platform in Titanium


    var Platform = require('org.favo.platform');
    
    // provides:
    Platform.isTablet
    Platform.isPhablet
    Platform.isPhone
    Platform.isAndroid
    Platform.isIos
    Platform.isIos7Plus
    Platform.iPhoneDevice
    Platform.isSimulator
	Platform.hasTouchId
